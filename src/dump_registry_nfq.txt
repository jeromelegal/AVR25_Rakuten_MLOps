#!/bin/bash
set -euo pipefail
HTTP_CODE=$(curl -k -o /dev/null -s -w "%{http_code}\n" https://$VAULT_SERVICE_NAME/health || echo 000)
until [ "$HTTP_CODE" -eq 200 ]; do
  echo "Waiting for Vault service to be healthy..."
  sleep 1
  HTTP_CODE=$(curl -k -o /dev/null -s -w "%{http_code}\n" https://$VAULT_SERVICE_NAME/health || echo 000)
done
export VAULT_SKIP_VERIFY="1"
mkdir -p $(dirname $REGISTRY_NFQ_PEM_PATH) $(dirname $REGISTRY_NFQ_CA_PATH)
until vault login -method=userpass username=$VAULT_USERNAME password=$VAULT_PASSWORD >/dev/null 2>&1; do
  echo "Vault auth failed. Retrying in 1s..."
  sleep 1
done
ips=$(ip -o -4 addr list | awk '{print $4}' | cut -d/ -f1 | paste -sd, -)
vault write -format=json pki_registry-nfq/issue/registry-nfq common_name="$SERVICE_NAME" ip_sans="$ips" alt_names="$SERVICE_NAME" ttl="72h" > /tmp/registry-nfq_cert.json || true
if [ -s /tmp/registry-nfq_cert.json ]; then
  jq -r '.data.certificate' /tmp/registry-nfq_cert.json > $REGISTRY_NFQ_CERT_PATH
  jq -r '.data.private_key' /tmp/registry-nfq_cert.json > $REGISTRY_NFQ_KEY_PATH
  jq -r '.data.issuing_ca'  /tmp/registry-nfq_cert.json > $REGISTRY_NFQ_CA_PATH
  cat <<EOF > $REGISTRY_NFQ_PEM_PATH
$(jq -r '.data.private_key' /tmp/registry-nfq_cert.json)
$(jq -r '.data.certificate' /tmp/registry-nfq_cert.json)
$(jq -r '.data.issuing_ca' /tmp/registry-nfq_cert.json)
EOF
  chown registry-nfq:registry-nfq $REGISTRY_NFQ_PEM_PATH || true
  chmod 400 $REGISTRY_NFQ_PEM_PATH || true
  cp $REGISTRY_NFQ_CA_PATH /usr/local/share/ca-certificates/registry-nfq.crt || true
  update-ca-certificates || true
else
  echo "WARN: Could not issue certificate from Vault PKI pki_registry-nfq."
fi

import time
import requests
import pytest

from conftest import run, http_get


@pytest.mark.integration
def test_proxy_is_up(proxy_endpoints, have_tools):
    # /health (fcgiwrap/nginx)
    r = http_get(f"{proxy_endpoints['proxy_https']}/health", verify=False)
    assert r.status_code == 200
    assert "Healthy" in r.text

    # /v2/ (root)
    r = http_get(f"{proxy_endpoints['proxy_https']}/v2/", verify=False)
    assert r.status_code == 200


@pytest.mark.integration
def test_nfqueue_pull_triggers_prefetch(proxy_endpoints):
    """
    Pull via le proxy d’une image depuis docker.io :
      - la requête de manifest passe par NFQUEUE,
      - le proxy prefetch l’image dans le tampon (internal-registry),
      - la requête est servie puis un second pull doit "hit".
    """
    repo = "library/busybox"
    tag = "latest"
    internal_tags_url = f"{proxy_endpoints['internal_registry']}/v2/{repo}/tags/list"

    # 0) Assure-toi que le tag n'est pas déjà présent dans le tampon
    r0 = requests.get(internal_tags_url, timeout=5)
    if r0.status_code == 200 and tag in (r0.json().get("tags") or []):
        # si déjà présent, on change de tag pour forcer le chemin NFQUEUE/prefetch
        repo = "library/alpine"
        tag = "latest"

    # 1) Pull via le proxy en source
    #    NB: référence 'docker://localhost/<repo>:<tag>' -> le proxy (TLS self-signed)
    proxy_ref = f"docker://localhost/{repo}:{tag}"
    run([
        "skopeo", "copy",
        "--retry-times", "3",
        "--src-tls-verify=false",
        proxy_ref,
        "dir:/tmp/proxy-pull-test"
    ])

    # 2) Poll du tampon jusqu'à présence du tag
    deadline = time.time() + 30
    ok = False
    while time.time() < deadline:
        r = requests.get(internal_tags_url, timeout=5)
        if r.status_code == 200 and tag in (r.json().get("tags") or []):
            ok = True
            break
        time.sleep(0.5)

    assert ok, "Prefetch NFQUEUE n’a pas peuplé le registry interne à temps"

    # 3) Second pull (hit)
    run([
        "skopeo", "copy",
        "--retry-times", "3",
        "--src-tls-verify=false",
        proxy_ref,
        "dir:/tmp/proxy-pull-test-2"
    ])


@pytest.mark.integration
def test_push_via_proxy_then_pull(proxy_endpoints):
    """
    Push d'un nouveau tag via le proxy (dest=proxy),
    qui doit router vers le tampon; puis pull via le proxy.
    """
    repo = "library/busybox"
    new_tag = "itestpush"

    # 1) Push via le proxy (dest TLS self-signed)
    dest_proxy = f"docker://localhost/{repo}:{new_tag}"
    src_public = "docker://docker.io/library/busybox:latest"
    run([
        "skopeo", "copy",
        "--retry-times", "3",
        src_public,
        dest_proxy,
        "--dest-tls-verify=false",
    ])

    # 2) Vérifie côté tampon
    internal_tags_url = f"{proxy_endpoints['internal_registry']}/v2/{repo}/tags/list"
    r = requests.get(internal_tags_url, timeout=5)
    assert r.status_code == 200
    assert new_tag in (r.json().get("tags") or [])

    # 3) Pull via le proxy (doit réussir)
    proxy_ref = f"docker://localhost/{repo}:{new_tag}"
    run([
        "skopeo", "copy",
        "--retry-times", "3",
        "--src-tls-verify=false",
        proxy_ref,
        "dir:/tmp/proxy-pull-itestpush"
    ])

import os
import time
import shutil
import subprocess
import requests
import pytest


def which_or_none(cmd):
    return shutil.which(cmd)

def run(cmd, check=True, capture_output=True, text=True, env=None):
    return subprocess.run(cmd, check=check, capture_output=capture_output, text=text, env=env)

def wait_http_ok(url, timeout=10, verify=False):
    deadline = time.time() + timeout
    while time.time() < deadline:
        try:
            r = requests.get(url, timeout=1, verify=verify)
            if r.status_code == 200:
                return True
        except Exception:
            pass
        time.sleep(0.25)
    return False

@pytest.fixture(scope="session")
def have_tools():
    # Vérifie que 'skopeo' est dispo dans l'image
    missing = []
    for tool in ("skopeo",):
        if not which_or_none(tool):
            missing.append(tool)
    if missing:
        pytest.skip(f"Missing tools in container: {', '.join(missing)}")
    return True

@pytest.fixture(scope="session")
def proxy_endpoints():
    """
    Par défaut on cible:
      - PROXY_BASE=https://localhost (proxy dans le conteneur)
      - INTERNAL_REGISTRY_BASE=http://internal-registry:5000 (service compose)
    Surchageables via variables d'env.
    """
    proxy_base = os.getenv("PROXY_BASE", "https://localhost").rstrip("/")
    internal_base = os.getenv("INTERNAL_REGISTRY_BASE", "http://internal-registry:5000").rstrip("/")
    return {
        "proxy_https": proxy_base,
        "internal_registry": internal_base,
    }

def http_get(url, **kw):
    return requests.get(url, timeout=kw.get("timeout", 5), verify=kw.get("verify", False))

#!/bin/bash

# Chemin du socket fcgiwrap
SOCKET_PATH="/var/run/fcgiwrap/fcgiwrap.socket"

# Créer le répertoire pour le socket si nécessaire
mkdir -p $(dirname "$SOCKET_PATH")

# Démarrer fcgiwrap en arrière-plan
/usr/sbin/fcgiwrap -s unix:$SOCKET_PATH &
FCGIWRAP_PID=$!

# Attendre que le socket soit créé
until [ -e "$SOCKET_PATH" ]; do
  echo "Attente de la création du socket $SOCKET_PATH..."
  sleep 1
done

# Définir les permissions pour le socket et le script de vérification de santé
chown nobody:nogroup $SOCKET_PATH /usr/local/bin/health-check.sh
chmod 660 $SOCKET_PATH /usr/local/bin/health-check.sh
chmod g+w $SOCKET_PATH /usr/local/bin/health-check.sh
chmod +x /usr/local/bin/health-check.sh


mkdir -p $(dirname $NGINX_STATUS_FILE)
mkdir -p $(dirname $NGINX_CONTENT_FILE)

echo "200 OK" > $NGINX_STATUS_FILE
echo "Healthy" > $NGINX_CONTENT_FILE

# Générer la configuration Nginx
nginx-conf.sh


# tail -f  /dev/null

# Tester la configuration Nginx
nginx -t

# Démarrer Nginx
nginx
NGINX_PID=$!

# Vérifier la santé du service
HTTP_CODE=$(curl -k -o /dev/null -s -w "%{http_code}\n" https://$SERVICE_NAME/health)

# Afficher le code de réponse
echo "HTTP Code: $HTTP_CODE"

# Vérifier si le service est sain
until [ $HTTP_CODE -ge 200 ] && [ $HTTP_CODE -lt 300 ]; do
  HTTP_CODE=$(curl -k -o /dev/null -s -w "%{http_code}\n" https://$SERVICE_NAME/health)
  # Afficher le code de réponse
  echo "HTTP Code: $HTTP_CODE"
  echo "Service is unhealthy."
  sleep 1
done


echo "Service HTTP CODE: $HTTP_CODE."

=== Contenu du fichier : ./nginx/nginx-conf.sh ===
#!/bin/bash



cat <<EOF > $NGINX_CONFIG_FILE
events { worker_connections 1024; }

# ➜ erreurs dans un fichier dédié
error_log  /var/log/nginx/registry-nfq.error.log info;

http {
    # ➜ access logs dans un fichier dédié
    log_format main '\$remote_addr - \$remote_user [\$time_local] "\$request" '
                    '\$status \$body_bytes_sent "\$http_referer" "\$http_user_agent" '
                    '\$request_time "\$upstream_addr" "\$upstream_status" "\$upstream_response_time"';
    access_log /var/log/nginx/registry-nfq.access.log main;


    server {
    listen 443 ssl;
    server_name $SERVICE_NAME;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_certificate $REGISTRY_NFQ_CERT_PATH;
    ssl_certificate_key $REGISTRY_NFQ_KEY_PATH;
    location /health {
        include fastcgi_params;
        fastcgi_pass unix:/var/run/fcgiwrap/fcgiwrap.socket;
        fastcgi_param SCRIPT_FILENAME $HEALTH_CHECK_FILE;
        fastcgi_param DOCUMENT_ROOT /usr/local/bin;
        fastcgi_param SCRIPT_NAME health-check.sh;
        fastcgi_param QUERY_STRING "";
    }
    location /v2/ {
        proxy_http_version 1.1;
        proxy_pass $PROXY_APP_BASE_URL;
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
    }
    }
}
EOF
# tail -f /dev/null

from netfilterqueue import NetfilterQueue
from scapy.all import IP, TCP, Raw
import threading, re

class NFQWorker:
    def __init__(self, queue_num: int = 0, listen_port: int = 5000, on_request=None):
        self.qn = queue_num
        self.port = listen_port
        self.nfq = NetfilterQueue()
        self.on_request = on_request
        self.req_regex = re.compile(rb"^(GET|HEAD)\s+([^\s]+)\s+HTTP/1\.[01]", re.M)

    def _cb(self, pkt):
        try:
            p = IP(pkt.get_payload())
            if p.haslayer(TCP) and p[TCP].dport == self.port and p.haslayer(Raw):
                data = bytes(p[Raw].load)
                m = self.req_regex.search(data)
                if m:
                    method = m.group(1).decode(errors="ignore")
                    path = m.group(2).decode(errors="ignore")
                    if self.on_request:
                        threading.Thread(target=self.on_request, args=(method, path), daemon=True).start()
        except Exception:
            pass
        pkt.accept()

    def start(self):
        t = threading.Thread(target=self._run, daemon=True)
        t.start()

    def _run(self):
        self.nfq.bind(self.qn, self._cb)
        try:
            self.nfq.run()
        finally:
            self.nfq.unbind()

import asyncio, os, json
import httpx

TAMPON_REGISTRY = os.getenv("TAMPON_REGISTRY", "127.0.0.1:5001")
REG_LOCAL_BASE_URL = os.getenv("REG_LOCAL_BASE_URL", "http://127.0.0.1:5001")
INSECURE_SRC = os.getenv("INSECURE_SRC", "false").lower() == "true"
INSECURE_DEST = os.getenv("INSECURE_DEST", "false").lower() == "true"

def skopeo_ref(ref: str) -> str:
    return f"docker://{ref}"

async def run(cmd: list[str], timeout: int = 1800) -> tuple[int, str, str]:
    p = await asyncio.create_subprocess_exec(*cmd, stdout=asyncio.subprocess.PIPE, stderr=asyncio.subprocess.PIPE)
    out, err = await p.communicate()
    return p.returncode, out.decode(), err.decode()

async def skopeo_inspect_digest(ref: str) -> str:
    cmd = ["skopeo", "inspect"]
    if INSECURE_SRC:
        cmd += ["--src-tls-verify=false"]
    cmd += [skopeo_ref(ref)]
    code, out, err = await run(cmd, timeout=120)
    if code != 0:
        raise RuntimeError(f"skopeo inspect failed for {ref}: {err.strip()}")
    info = json.loads(out)
    digest = info.get("Digest")
    if not digest or not digest.startswith("sha256:"):
        raise RuntimeError(f"Cannot resolve digest for {ref}")
    return digest

async def local_has_manifest(reg: str, name: str, digest: str) -> bool:
    url = f"{REG_LOCAL_BASE_URL}/v2/{reg}/{name}/manifests/{digest}"
    async with httpx.AsyncClient(timeout=10) as client:
        r = await client.head(url, headers={
            "Accept":"application/vnd.oci.image.manifest.v1+json, application/vnd.docker.distribution.manifest.v2+json"
        })
        return r.status_code == 200

async def copy_to_local(reg: str, name: str, digest: str) -> None:
    src = f"{reg}/{name}@{digest}"
    dst = f"{TAMPON_REGISTRY}/{reg}/{name}@{digest}"
    cmd = ["skopeo", "copy", "--all"]
    if INSECURE_SRC:
        cmd += ["--src-tls-verify=false"]
    if INSECURE_DEST:
        cmd += ["--dest-tls-verify=false"]
    cmd += [skopeo_ref(src), skopeo_ref(dst)]
    code, out, err = await run(cmd, timeout=1800)
    if code != 0:
        raise RuntimeError(f"skopeo copy failed: {err.strip()}")

app = FastAPI()

# Headers hop-by-hop à exclure
_HOP = {
    "connection","keep-alive","proxy-authenticate","proxy-authorization",
    "te","trailer","transfer-encoding","upgrade","host"
}
def _filter_headers(h: dict) -> dict:
    return {k: v for k, v in h.items() if k.lower() not in _HOP}

# Accept pour manifests (schema2 / list / oci)
_MANIFEST_ACCEPT = (
    "application/vnd.docker.distribution.manifest.v2+json,"
    "application/vnd.docker.distribution.manifest.list.v2+json,"
    "application/vnd.oci.image.index.v1+json,"
    "application/vnd.oci.image.manifest.v1+json,"
    "application/json"
)

def parse_manifest_path(path: str):
    m = re.match(r"^/v2/([^/]+)/(.+)/manifests/([^/]+)$", path)
    return m.groups() if m else None

def parse_blob_path(path: str):
    m = re.match(r"^/v2/([^/]+)/(.+)/blobs/(sha256:[0-9a-f]{64})$", path)
    return m.groups() if m else None

async def _head_local_manifest_digest(reg: str, name: str, ref: str) -> str | None:
    """HEAD sur le tampon pour récupérer Docker-Content-Digest d’un tag (ou digest)."""
    url = f"{REG_LOCAL_BASE_URL}/v2/{reg}/{name}/manifests/{ref}"
    async with httpx.AsyncClient(timeout=None) as client:
        r = await client.head(url, headers={"Accept": _MANIFEST_ACCEPT})
    if r.status_code == 200:
        dcd = r.headers.get("Docker-Content-Digest")
        if dcd and dcd.startswith("sha256:"):
            return dcd
    return None

async def ensure_local(reg: str, name: str, ref: str) -> str:
    """
    Garantit que le manifest référencé par 'ref' est présent localement
    et renvoie le **digest tel qu’il existe dans le tampon**.
    - Si ref est un digest -> on s’assure juste qu’il est présent.
    - Si ref est un tag -> on préfère toujours le digest **local** du tag.
    """
    # Cas ref = digest
    if ref.startswith("sha256:"):
        key = f"{reg}/{name}@{ref}"
        async with ensure_lock:
            inflight.setdefault(key, asyncio.Lock())
        async with inflight[key]:
            if await local_has_manifest(reg, name, ref):
                METRICS["ensures_hit"] += 1
                return ref
            METRICS["ensures_miss"] += 1
            async with sem:
                await copy_to_local(reg, name, ref)
            return ref

    # Cas ref = tag : d’abord, regard local
    local_d = await _head_local_manifest_digest(reg, name, ref)
    if local_d:
        METRICS["ensures_hit"] += 1
        return local_d

    # Sinon, on résout le digest externe (manif/list) puis on copie
    ext_digest = await skopeo_inspect_digest(f"{reg}/{name}:{ref}")
    key = f"{reg}/{name}@{ext_digest}"
    async with ensure_lock:
        inflight.setdefault(key, asyncio.Lock())
    async with inflight[key]:
        if await local_has_manifest(reg, name, ext_digest):
            # Après présence, récupère le digest réellement servi localement pour ce tag
            final_d = await _head_local_manifest_digest(reg, name, ref) or ext_digest
            METRICS["ensures_hit"] += 1
            return final_d
        METRICS["ensures_miss"] += 1
        async with sem:
            await copy_to_local(reg, name, ext_digest)
        # Important : renvoyer le digest **local** du tag (peut différer de ext_digest si list vs image)
        final_d = await _head_local_manifest_digest(reg, name, ref) or ext_digest
        return final_d

    out, err = await p.communicate()
    return p.returncode, out.decode(), err.decode()

async def skopeo_inspect_digest(ref: str) -> str:
    cmd = ["skopeo", "inspect"]
    if INSECURE_SRC:
        cmd += ["--src-tls-verify=false"]
    cmd += [skopeo_ref(ref)]
    code, out, err = await run(cmd, timeout=120)
    if code != 0:
        raise RuntimeError(f"skopeo inspect failed for {ref}: {err.strip()}")
    info = json.loads(out)
    digest = info.get("Digest")
    if not digest or not digest.startswith("sha256:"):
        raise RuntimeError(f"Cannot resolve digest for {ref}")
    return digest

async def local_has_manifest(reg: str, name: str, digest: str) -> bool:
    url = f"{REG_LOCAL_BASE_URL}/v2/{reg}/{name}/manifests/{digest}"
    async with httpx.AsyncClient(timeout=10) as client:
        r = await client.head(url, headers={
            "Accept":"application/vnd.oci.image.manifest.v1+json, application/vnd.docker.distribution.manifest.v2+json"
        })
        return r.status_code == 200

async def copy_to_local(reg: str, name: str, digest: str) -> None:
    src = f"{reg}/{name}@{digest}"
    dst = f"{TAMPON_REGISTRY}/{reg}/{name}@{digest}"
    cmd = ["skopeo", "copy", "--all"]
    if INSECURE_SRC:
        cmd += ["--src-tls-verify=false"]
    if INSECURE_DEST:
        cmd += ["--dest-tls-verify=false"]
    cmd += [skopeo_ref(src), skopeo_ref(dst)]
    code, out, err = await run(cmd, timeout=1800)
    if code != 0:
        raise RuntimeError(f"skopeo copy failed: {err.strip()}")
